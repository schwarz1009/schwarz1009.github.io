struct _obj {
    int p;
};

int check1 (struct _obj *obj) {
    if (obj->p == 0) {
        return 1;
    }
    return 0;
}
int check2 (struct _obj *obj) {
    if (obj->p == 1) {
        return 1;
    }
    return 0;
}
void action1 (struct _obj *obj) {
    printf("ここではアクション１をやります。\n");
}
void action2 (struct _obj *obj) {
    printf("ここではアクション２をやります。\n");
}
int main (void) {
    struct _obj obj;
    obj.p = 0;
    if (check1(&obj)){
        action1(&obj);
    }
    if (check2(&obj)){
        action2(&obj);
    }
}
